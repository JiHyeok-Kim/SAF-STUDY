#include <iostream>
using namespace std;

int main()
{
	int vect[3][5] =
	{
		5, 1, 4, 2, 6,
		3, 5, 0, 0, 7,
		9, 9, 8, 3, 1,
	};

	int x, y;
	int input;
	int cnt = 0;

	cin >> input;

	for (y = 0; y < 3; y++)
	{
		for (x = 0; x < 5; x++)
		{
			if (vect[y][x] > input)
			{
				cnt++;
			}
		}
	}

	cout << cnt << "°³";


	return 0;
}
